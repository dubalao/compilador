program TesteFuncoes2;

(* Objetivo: Testar a declaração e chamada de uma função, com variáveis e expressões. *)
var
    valor_recebido : integer;
    numero1 : integer;
    numero2 : integer;
    resultado : integer;

function calcula_valor(val1 : integer; val2 : integer) : integer;
begin
    calcula_valor := val1 + val2  { Retorna a soma dos dois valores }
end;

function calcula_dobro(val : integer) : integer;
begin
    calcula_dobro := val * 2  { Retorna o dobro do valor }
end;

begin
    numero1 := 10;
    numero2 := 5;
    
    valor_recebido := calcula_valor(numero1, numero2);  { Chama a função 'calcula_valor' }
    
    resultado := calcula_dobro(valor_recebido);  { Chama a função 'calcula_dobro' com o valor retornado }

    write(resultado)  { Exibe o valor do resultado }
end.
